```
function alphabetPosition(text)
{
  text = text.toLowerCase();
  let new_str;
  for (let char of text)
  {
    if (char >= 'a' && char <= 'z')
    {
      let nb = char.charCodeAt(0) - 96;
      if (!new_str)
        new_str = nb;
      else
        new_str = new_str + " " + nb;
    }
  }
  if (!new_str)
    new_str = '';
  return String(new_str);
}
```
